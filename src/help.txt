Help ModalView

import Tutorial;

$ctn: generateElement('div'),
$heading: generateEleent('h1'),
$description: generateElement('div'),
$tutorialBtn: generateButton,
$grsBtn: generateButton,
$imagesBtn: generateButton,

init (modal):
  this.modal = modal
  this.tutorial = Object.create(Tutorial);
  $tutorialBtn.onclick = startTutorial;
  $grsBtn.onclick = displayGRSTutorial;
  $imagesBtn.onclick = displayImagesTutorial;

startTutorial:
  this.modal.hide;
  this.tutorial.startTutorial;

setModalBackBtn:
  this.modal.setSaveHandler('Back', this.display);

displayGRSTutorial:
  this.setModalBackBtn;
  this.modal.display($grsTutorial);

displayImagesTutorial:
  this.setModalBackBtn;
  this.modal.display($imagesTutorial);
